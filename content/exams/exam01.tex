\section*{Ⅰ. 单选题}\begin{question}
设复数 $(1+5i)i$ 的虚部为（ ）。

\begin{choices}
  \choice $-1$
  \choice $0$
  \choice $1$
  \choice $5$
\end{choices}

\topics{复数; 虚部计算}
\difficulty{0.3}
\explain{直接相乘：$(1+5i)i = i + 5i^2 = i - 5$，虚部为 $1$。}
\end{question}

\begin{question}
若指数函数 $f(x)=2^x$，则 $f(0)+f(1)$ 等于（ ）。 


\answer{ex2}{答案：$1+2=3$。}
