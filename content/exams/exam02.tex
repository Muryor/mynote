\section*{Ⅰ. 单选题}

\begin{question}[points=2]
已知集合 $A=\{x\mid \log_2 x < 1\},\, B=\{x\mid x<1\}$，则 $A\cap B$ 等于（ ）
\begin{choices}[columns=2]
  \choice $(-\infty,1)$
  \CorrectChoice $(0,1)$
  \choice $(-\infty,2)$
  \choice $(0,2)$
\end{choices}
\topics{交集；不等式与函数单调性}
\difficulty{0.40}
\explain{由 $\log_2 x<1\Rightarrow 0<x<2$，与 $x<1$ 取交得 $(0,1)$。}
\source{自编}
\end{question}
