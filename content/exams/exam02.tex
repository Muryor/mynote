\section*{Ⅰ. 单选题}

\begin{question}
已知集合 $A=\{x\mid \log_2 x < 1\},\, B=\{x\mid x<1\}$，则 $A\cap B$ 等于 \paren[B]
\begin{choices}
  \item $(-\infty,1)$
  \item $(0,1)$
  \item $(-\infty,2)$
  \item $(0,2)$
\end{choices}
\end{question}
\topics{交集；不等式与函数单调性}
\difficulty{0.40}
\explain{由 $\log_2 x<1\Rightarrow 0<x<2$，与 $x<1$ 取交得 $(0,1)$。}
\source{自编}

\begin{question}
已知函数 $f(x)=2^x$，则 $f(0)+f(1)$ 等于 \paren[C]
\begin{choices}
  \item $1$
  \item $2$
  \item $3$
  \item $4$
\end{choices}
\end{question}
\topics{指数函数；基本运算}
\difficulty{0.20}
\explain{$f(0)=2^0=1$，$f(1)=2^1=2$，故 $f(0)+f(1)=3$。}


\begin{question}
已知向量$\vec{a}=(0,1),\vec{b}=(2,x)$ ，若$\vec{b}\perp(\vec{b}-4\vec{a})$ ，则x=（）\paren[C]
\begin{choices}
 \item $1$
 \item $-2$
 \item $-1$
 \item $2$
\end{choices}
\topics{平面向量线性运算的坐标表示；向量垂直的坐标表示}
\difficulty{0.85}
\explain{
根据向量垂直的坐标运算可求$x$ 的值
因为$\vec{b}\perp\left(\vec{b}-4\vec{a}\right)$ ，所以$\vec{b}\cdot\left(\vec{b}-4\vec{a}\right)=0$ 所以$\vec{b}^{2}-4\vec{a}\cdot\vec{b}=0$ 即$4+x^{2}-4x=0$ ，故$x=2$ 故选：D.
本题答案：D。
}\end{question}

\begin{question}
已知函数$f(x)=\begin{cases}-x^{2}-2ax-a,x<0\\\mathrm{e}^{x}+\ln(x+1),x\geq0\end{cases}$ 在$R$ 上单调递增，则$a$ 的取值范围是（）
\begin{choices}
\item  $(∞0,0]$
\item  $[0,+\infty)$
\item  $[-1,0]$
\item   $[-1,1]$
\end{choices}
\topics{判断指数函数的单调性；研究对数函数的单调性；根据分段函数的单调性求参数}
\difficulty{0.65}
\explain{
根据二次函数的性质和分界点的大小关系即可得到不等式组，解出即可
因为$f(x)$ 在R上单调递增，且$x\geq0$ 时， $f(x)=\mathrm{e}^{x}+\ln\left(x+1\right)$ 单调递增， 周质润足$\begin{cases}-\frac{-2a}{2\times\left(-1\right)}\geq0\end{cases}$ $\left\lfloor-a\leq\mathrm{e}^{0}+\ln1\right\rfloor$ 解拍$-1\leq a\leq0$ 即$a$ 的范围是[-1,0] 故选：B.
本题答案：B。
}\end{question}

\begin{question}
  函数 $f(x)=x^2-4x+3$ 的最小值为 \fillin[-1]{}。
  \topics{二次函数；顶点式}
  \difficulty{0.40}
