\chapter{椭圆及其标准方程}

\begin{definitionx}{椭圆的定义}{def:ellipse}
平面内与两个定点 $F_1,F_2$ 的距离之和等于常数（大于 $|F_1F_2|$）的点的轨迹称为\textbf{椭圆}。
该常数记为 $2a$，两焦点间距为 $2c$，离心率 $e=\dfrac{c}{a}(0<e<1)$。
\end{definitionx}

\begin{propertyx}{标准方程与几何量}{prop:std-eq}
当椭圆中心在原点、焦点在 $x$ 轴上时，其标准方程为
\[
\frac{x^2}{a^2}+\frac{y^2}{b^2}=1,\qquad a>b>0,\quad c^2=a^2-b^2.
\]
长轴长 $2a$、短轴长 $2b$，焦距 $2c$。
\end{propertyx}

\begin{examplex}{例题}{ex:01}
已知椭圆 $C:\,\dfrac{x^2}{a^2}+\dfrac{y^2}{b^2}=1\ (a>b>0)$ 的左、右焦点分别为 $F_1,F_2$。
求椭圆 $C$ 的方程。
\explain{略。可由定义 $|PF_1|+|PF_2|=2a$ 推导，或由几何关系 $c^2=a^2-b^2$ 确定参数。}
\end{examplex}

\begin{notebox}{}
记忆：离心率 $e\to0$ 椭圆 $\to$ 圆；$e\to1$ 椭圆 $\to$ 近似线段。
\end{notebox}
